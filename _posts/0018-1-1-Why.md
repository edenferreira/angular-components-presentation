### Testing
```javascript
describe('navigable', function () {
  var navigableCtrl;
  var navigationMock;

  beforeEach(module('navigation'));
  beforeEach(inject(function ($componentController) {
    navigableCtrl = $componentController('navigable');
    navigationMock = {};
    navigableCtrl.navigation = navigationMock;
  }));

  it('when the mock returns true', function () {
    navigationMock.isActive = () => true;
    expect(navigableCtrl.isActive()).toBeTruthy();
  });

  it('when the mock returns false', function () {
    navigationMock.isActive = () => false;
    expect(navigableCtrl.isActive()).toBeFalsy();
  });
});
```
