### Testing
```javascript
describe('navigation', function () {
  var navigationCtrl;
  beforeEach(module('navigation'));
  beforeEach(inject(function ($componentController) {
    navigationCtrl = $componentController('navigation');
  }))
  
  it('when I check if is active', function () {
    expect(navigationCtrl.isActive()).toBeFalsy();
  })
  
  describe('given a new item', function () {
    var newItem;
    beforeEach(function () {
      newItem = 'new name';
    });
    
    it('when I add it', function () {
      navigationCtrl.addNewItem(newItem);
      expect(navigationCtrl.isActive(newItem)).toBeTruthy();
    });
  });
});
```
