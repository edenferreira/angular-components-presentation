### Testing
```javascript
describe('navigable', function () {
    var navigableCtrl,
        navigationMock = {};
    beforeEach(module('navigation'));
    beforeEach(inject(function ($componentController) {
        navigableCtrl =  $componentController('navigable');
    }));

    describe('given a mock required navigation', function () {
        beforeEach(function () {
            navigationMock = {
                addNewItem: function () {
                    return 5
                },
                isActive: function (num) {
                    return num === 5;
                }
            };
            navigableCtrl.navigation = navigationMock;
        });

        it('when The component is initiated and I check if is active', function () {
            navigableCtrl.$onInit();
            expect(navigableCtrl.isActive()).toBe(true);
        });
    });
});
```
